model{
  m <- round(sum(M[] * phi[])) #marks available for recapture
  Nm <- round(N - m)
  PHI <- C / N
  for(i in 1:2){
	R[i] ~ dbinom(phi0[i], M[i])
	phi0[i] <- PHI * phi[i]
#	phi[i] ~ dbeta(1,1)
  }
  phi[1] ~ dunif(0, 1)
  phi[2] ~ dunif(0, phi[1])
  Rsum ~ dhyper(m, Nm, C, 1)
  N ~ dunif(0, 25000)
}
